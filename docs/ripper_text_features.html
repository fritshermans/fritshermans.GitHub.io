from sklearn.base import TransformerMixin, BaseEstimator
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import make_pipeline


class TopTokenTransformer(TransformerMixin, BaseEstimator):
    def __init__(self, col, top_n=5, vectorizer_kwargs={}):
        self.col = col
        self.top_n = top_n
        self.vectorizer_kwargs = vectorizer_kwargs
        
    @staticmethod
    def get_top_tokens(feature_names, coefs, top_n):
        # tokens = pd.DataFrame({'token': feature_names, 'coef':coefs})
        # tokens['abs_coef'] = tokens['coef'].abs()
        # return (tokens.sort_values('abs_coef', ascending=False)
        #         .head(top_n)['token'].tolist())
        return [x for x, _ in sorted(zip(feature_names, coefs), key=lambda pair: abs(pair[1]), reverse=True)]
    
    def fit(self, X, y):
        pipe = make_pipeline(CountVectorizer(**self.vectorizer_kwargs),
                            LogisticRegression(class_weight='balanced', max_iter=1000))
        pipe.fit(X[self.col],y)
        vec = pipe.steps[0][1]
        clf = pipe.steps[1][1]
        feature_names = vec.get_feature_names_out()
        coefs = clf.coef_[0]
        self.top_tokens_ = TopTokenTransformer.get_top_tokens(feature_names, coefs, self.top_n)
        self.vec_top = CountVectorizer(vocabulary=self.top_tokens_, **self.vectorizer_kwargs)
        return self
    
    def transform(self, X):
        return self.vec_top.transform(X[self.col]).toarray()
    
    def get_feature_names_out(self, input_features=None):
        return self.top_tokens_
